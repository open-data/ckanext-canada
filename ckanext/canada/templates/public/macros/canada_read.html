{%- endmacro -%} {# Render a list of tag vocabulary tags for given field object and client language #} {%- macro tag_vocab_field(f, pkg, client_lang) -%}
    {%- for tag in pkg[f.id] -%}
  • {{ tag.split(' ')[1 if client_lang == 'fr' else 0] }}
    • {%- endfor -%}
